        if(idx===-1) y.push(user.n); else y.splice(idx,1);

        await sb.from('posts').update({yeahs: y}).eq('id', id);

        renderPosts();

    }


    async function startDM(partner) { activeChat = partner; document.getElementById('chat-with-name').innerText = "Czat z: " + partner; showSec('chat'); renderMessages(); }

    async function sendDM() {

        const t = document.getElementById('chat-input').value.trim(); if(!t) return;

        await sb.from('messages').insert([{sender:user.n, receiver:activeChat, content:t}]);

        document.getElementById('chat-input').value = ""; renderMessages();

    }

    async function renderMessages() {

        const win = document.getElementById('chat-window'); win.innerHTML = "";

        let {data: msgs} = await sb.from('messages').select('*').or(`and(sender.eq.${user.n},receiver.eq.${activeChat}),and(sender.eq.${activeChat},receiver.eq.${user.n})`).order('id', {ascending:true});

        if(msgs) msgs.forEach(m => { win.innerHTML += `<div class="chat-msg ${m.sender===user.n?'msg-sent':'msg-rec'}">${m.content}</div>`; });

        win.scrollTop = win.scrollHeight;

    }

    async function renderInbox() {

        const h = document.getElementById('dm-inbox-list'); h.innerHTML = "Ładowanie...";

        let {data: msgs} = await sb.from('messages').select('*').or(`sender.eq.${user.n},receiver.eq.${user.n}`);

        const partners = new Set();

        if(msgs) msgs.forEach(m => { if(m.sender===user.n) partners.add(m.receiver); else partners.add(m.sender); });

        h.innerHTML = partners.size ? "" : "Brak rozmów.";

        partners.forEach(p => { h.innerHTML += `<div class="post" style="cursor:pointer" onclick="startDM('${p}')"><b>${p}</b></div>`; });

    }


    async function viewUserProfile(name) {

        let {data:u} = await sb.from('profiles').select('*').eq('username', name).single();

        if(!u) return uiNotify("Użytkownik nie istnieje");

        let {data:pCount} = await sb.from('posts').select('id', { count: 'exact' }).eq('author', name);

        const isFol = (u.followers || []).includes(user.n);

        

        document.getElementById('user-profile-content').innerHTML = `

            <img src="${u.avatar}" style="width:100px; height:100px; border-radius:25px; border:3px solid white; box-shadow:0 4px 10px rgba(0,0,0,0.1)">

            <h2 style="margin:10px 0;">${name}</h2>

            <div class="stats-grid">

                <div class="stat-card"><span class="stat-num">${pCount ? pCount.length : 0}</span>Wpisy</div>

                <div class="stat-card"><span class="stat-num">${(u.followers||[]).length}</span>Obs.</div>

            </div>

            <div style="display:flex; gap:10px; justify-content:center; margin-top:15px;">

                ${name !== user.n ? `

                    <button class="btn btn-follow ${isFol?'active':''}" onclick="toggleFollow('${name}')">${isFol?'Obserwujesz':'Obserwuj'}</button> 

                    <button class="btn btn-blue" onclick="startDM('${name}')">Wyślij DM</button>

                ` : '<p style="color:#888; font-style:italic;">To Twój profil</p>'}

            </div>

        `;

        showSec('user-view');

    }


    async function toggleFollow(name) {

        let {data:u} = await sb.from('profiles').select('followers').eq('username', name).single();

        let f = u.followers || [];

        const idx = f.indexOf(user.n);

        if(idx===-1) f.push(user.n); else f.splice(idx,1);

        await sb.from('profiles').update({followers: f}).eq('username', name);

        viewUserProfile(name);

    }


    async function renderReports() {

        let {data:reps} = await sb.from('reports').select('*');

        const list = document.getElementById('reports-list');

        list.innerHTML = reps && reps.length ? "" : "Brak zgłoszeń.";

        if(reps) reps.forEach(r => {

            list.innerHTML += `<div style="background:white; padding:10px; margin-bottom:5px; border-radius:10px; border:1px solid #ddd; font-size:12px;">

                <b>Od:</b> ${r.from_user} | <b>Post ID:</b> ${r.post_id}<br>

                <div style="display:flex; gap:5px; margin-top:5px;">

                    <button class="btn btn-red" style="padding:2px 10px; font-size:10px;" onclick="deleteReport(${r.id})">Usuń raport</button>

                </div>

            </div>`;

        });

    }


    async function goToReportedPost(postId) {

        let { data: post } = await sb.from('posts').select('category').eq('id', postId).single();

        if(post) {

            closeModals();

            showSec('feed');

            renderPosts(postId);

        } else {

            uiNotify("Ten post już nie istnieje.");

        }

    }
