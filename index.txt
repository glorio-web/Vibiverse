    function getCoords(e) {
        const rect = canvas.getBoundingClientRect();
        const clientX = e.clientX || (e.touches ? e.touches[0].clientX : 0);
        const clientY = e.clientY || (e.touches ? e.touches[0].clientY : 0);
        return { x: (clientX - rect.left) * (canvas.width / rect.width), y: (clientY - rect.top) * (canvas.height / rect.height) };
    }

        showSec('user-view');

    async function toggleFollow(name) {
        let {data:u} = await sb.from('profiles').select('followers').eq('username', name).single();
        let f = u.followers || [];
        const idx = f.indexOf(user.n);
        if(idx===-1) f.push(user.n); else f.splice(idx,1);
        await sb.from('profiles').update({followers: f}).eq('username', name);
        viewUserProfile(name);
    }

    async function renderReports() {
        let {data:reps} = await sb.from('reports').select('*');
        const list = document.getElementById('reports-list');
        list.innerHTML = reps && reps.length ? "" : "Brak zgłoszeń.";
        if(reps) reps.forEach(r => {
            list.innerHTML += `<div style="background:white; padding:10px; margin-bottom:5px; border-radius:10px; border:1px solid #ddd; font-size:12px;">
                <b>Od:</b> ${r.from_user} | <b>Post ID:</b> ${r.post_id}<br>
                <div style="display:flex; gap:5px; margin-top:5px;">
                    <button class="btn btn-blue" style="padding:2px 10px; font-size:10px;" onclick="goToReportedPost(${r.post_id})">Idź do posta</button>
                    <button class="btn btn-red" style="padding:2px 10px; font-size:10px;" onclick="deleteReport(${r.id})">Usuń raport</button>
                </div>
            </div>`;
        });
        openModal('modal-reports');
    }

    async function goToReportedPost(postId) {
        let { data: post } = await sb.from('posts').select('category').eq('id', postId).single();
        if(post) {
            closeModals();
            currentComm = post.category;
            document.getElementById('feed-title').innerText = post.category;
            showSec('feed');
            renderPosts(postId);
        } else {
            uiNotify("Ten post już nie istnieje.");
        }
    }

    async function reportPost(id) { await sb.from('reports').insert([{from_user:user.n, post_id:id}]); uiNotify("Zgłoszono."); }
    async function deleteReport(id) { await sb.from('reports').delete().eq('id', id); renderReports(); }

    async function updateAvatar() {
        const f = document.getElementById('edit-av-input').files[0]; if(!f) return;
        const r = new FileReader(); r.onload=async (e)=>{ await sb.from('profiles').update({avatar: e.target.result}).eq('username', user.n); location.reload(); }; r.readAsDataURL(f);
    }

    function deletePost(id) {
        document.getElementById('ui-title').innerText = "Usuwanie wpisu";
        document.getElementById('ui-msg').innerText = "Czy na pewno chcesz usunąć ten wpis?";
        document.getElementById('ui-btns').innerHTML = `
            <button class="btn" onclick="closeModals()">Nie</button>
            <button class="btn btn-red" onclick="finalDeletePost(${id})">Tak</button>
        `;
        openModal('ui-modal');
    }

    async function finalDeletePost(id) {
        await sb.from('posts').delete().eq('id', id);
        renderPosts();
        closeModals();
    }

    function confirmDelete() {
        document.getElementById('ui-title').innerText = "USUWANIE KONTA";
        document.getElementById('ui-msg').innerText = "Czy na pewno chcesz bezpowrotnie usunąć konto?";
        document.getElementById('ui-btns').innerHTML = `
            <button class="btn" onclick="closeModals()">Nie</button>
            <button class="btn btn-red" onclick="finalDeleteUser()">TAK</button>
        `;
        openModal('ui-modal');
    }

    async function finalDeleteUser() {
        await sb.from('profiles').delete().eq('username', user.n);
        location.reload();
    }

    function enterComm(n) { currentComm=n; document.getElementById('feed-title').innerText=n; showSec('feed'); renderPosts(); }
    
    function showSec(s) {
        document.querySelectorAll('.screen').forEach(x => x.classList.remove('active')); 
        document.getElementById('sec-'+s).classList.add('active');
        document.querySelectorAll('.nav-tab').forEach(t => t.classList.remove('active')); 
        const tab = document.getElementById('t-'+s.substring(0,4)); 
        if(tab) tab.classList.add('active');
        
        document.getElementById('write-wrap').style.display = (s==='feed'?'flex':'none');
        if(s==='profile') {
            document.getElementById('p-name').innerText = user.n; 
            document.getElementById('p-av-box').innerHTML = `<img src="${user.av}" style="width:100px; height:100px; border-radius:25px; border:4px solid white; box-shadow:0 4px 10px rgba(0,0,0,0.1)">`;
            renderMyStats();
        }
        if(s==='messages') renderInbox();
    }

    async function renderMyStats() {
        let {data:p} = await sb.from('posts').select('id', {count:'exact'}).eq('author', user.n);
        let {data:u} = await sb.from('profiles').select('followers').eq('username', user.n).single();
        document.getElementById('my-stats').innerHTML = `<div class="stat-card"><span class="stat-num">${p ? p.length : 0}</span>Wpisy</div><div class="stat-card"><span class="stat-num">${u && u.followers ? u.followers.length : 0}</span>Obs.</div>`;
    }

    function openModal(id) { 
        document.getElementById(id).style.display='flex'; 
        if(id==='modal-post'){ 
            canvas.width = canvas.offsetWidth; 
            canvas.height = canvas.offsetHeight; 
            if(!hasDraw) ctx.clearRect(0,0,canvas.width,canvas.height); 
        } 
    }
